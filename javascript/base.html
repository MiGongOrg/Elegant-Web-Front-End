Function

ES 5 VS ES 6 return sum

Before

var sum = function (x) {
  return x + x;
};
console.log(sum(2)); // 4

After

This is ES 6 code, you can also see the ES 5 output

ES 6
ES 5
let sum = (x) => { x + x }
console.log(sum(2)); // 4
"use strict";

var sum = function sum(x) {
  x + x;
};
console.log(sum(2)); // 4

Other

This is ES 6 code, you can also see the ES 5 output

ES 6
ES 5
let sum = x => x + x
console.log(sum(2)); // 4
"use strict";

var sum = function sum(x) {
  return x + x;
};
console.log(sum(2)); // 4

Array loops

Before

var array = [0, 1, 2, 3, 4, 5];
for (var i = 0; i < array.length; i++) {
  var item = array[i]
  console.log(item); // 0 1 2 3 4 5
};

After

var array = [0, 1, 2, 3, 4, 5];
array.forEach(function (item) {
  console.log(item); // 0 1 2 3 4 5
});

Other

This is ES 6 code, you can also see the ES 5 output

ES 6
ES 5
let array = [0, 1, 2, 3, 4, 5];
array.forEach(item => console.log(item)); // 0 1 2 3 4 5
"use strict";

var array = [0, 1, 2, 3, 4, 5];
array.forEach(function (item) {
  return console.log(item);
}); // 0 1 2 3 4 5
