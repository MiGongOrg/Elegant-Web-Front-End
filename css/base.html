Keyframe

Before

@keyframes down {
  0% {
    top: 0;
  }
  100% {
    top: 100%;
  }
}

After

@keyframes down {
  from {
    top: 0;
  }
  to {
    top: 100%;
  }
}

Other

@keyframes down {
  to {
    top: 100%;
  }
}

Alternate

Before

.animation {
  animation-name: down;
  animation-duration: 5s;
  animation-timing-function: linear;
  animation-iteration-count: infinite;
  animation-direction: alternate;
}
@keyframes down {
  0% {
    top: 0;
  }
  100% {
    top: 100%;
  }
}

After

.animation {
  animation: down 5s linear infinite alternate;
}
@keyframes down {
  to {
    top: 100%;
  }
}

Other

.animation {
  animation: down 5s linear infinite;
}
@keyframes down {
  50% {
    top: 100%;
  }
}
