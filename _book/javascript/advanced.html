String To Array

Before

var str = 'hello world';
console.log(str.split('')); // ["h", "e", "l", "l", "o", " ", "w", "o", "r", "l", "d"]

After

ES 5
let str = 'hello world';
console.log([...str]); // ["h", "e", "l", "l", "o", " ", "w", "o", "r", "l", "d"]
'use strict';

function _toConsumableArray(arr) { if (Array.isArray(arr)) { for (var i = 0, arr2 = Array(arr.length); i < arr.length; i++) { arr2[i] = arr[i]; } return arr2; } else { return Array.from(arr); } }

var str = 'hello world';
console.log([].concat(_toConsumableArray(str))); // ["h", "e", "l", "l", "o", " ", "w", "o", "r", "l", "d"]

Array Dedupe

Before

var array = [1, 1, 2, 2, 3, 3];
var af = array.filter(function (el, i, a) {
  return i === a.indexOf(el)
});
console.log(af); // [1, 2, 3]

After

ES 6
ES 5
let array = [1, 1, 2, 2, 3, 3];
console.log([...new Set(array)]); // [1, 2, 3]
"use strict";

function _toConsumableArray(arr) { if (Array.isArray(arr)) { for (var i = 0, arr2 = Array(arr.length); i < arr.length; i++) { arr2[i] = arr[i]; } return arr2; } else { return Array.from(arr); } }

var array = [1, 1, 2, 2, 3, 3];
console.log([].concat(_toConsumableArray(new Set(array)))); // [1, 2, 3]
